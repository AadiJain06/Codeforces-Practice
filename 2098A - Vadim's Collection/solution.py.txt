import sys

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    t = int(input_data[0])
    results = []
    
    for k in range(1, t + 1):
        s = sorted(input_data[k]) # Sort digits to easily pick the smallest
        n = 10
        res = [0] * n
        
        # We need to fill positions 0 to 9 (1st to 10th digit)
        for i in range(n):
            lower_bound = 9 - i
            # Find the smallest digit in 's' that is >= lower_bound
            for idx, digit in enumerate(s):
                if int(digit) >= lower_bound:
                    # Check if the remaining digits can satisfy remaining bounds
                    # Since the original was beautiful, and bounds decrease,
                    # picking the smallest valid digit here is optimal.
                    res[i] = digit
                    s.pop(idx)
                    break
        
        results.append("".join(res))
        
    sys.stdout.write("\n".join(results) + "\n")

solve()
