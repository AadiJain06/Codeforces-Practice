import sys
import math

# Helper function to find GCD of a list
def get_gcd(numbers):
    if not numbers:
        return 0
    res = numbers[0]
    for i in range(1, len(numbers)):
        res = math.gcd(res, numbers[i])
    return res

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    
    t = int(data[0])
    ptr = 1
    
    for _ in range(t):
        n = int(data[ptr])
        a = list(map(int, data[ptr+1 : ptr+1+n]))
        ptr += n + 1
        
        found = False
        for i in range(n):
            # Sequence C is just a[i]
            gcd_c = a[i]
            
            # Sequence B is everything else
            others = a[:i] + a[i+1:]
            gcd_b = get_gcd(others)
            
            if gcd_b != gcd_c:
                print("Yes")
                # Output 2 for sequence C at index i, 1 for others
                res = [1] * n
                res[i] = 2
                print(*(res))
                found = True
                break
        
        if not found:
            print("No")

solve()
