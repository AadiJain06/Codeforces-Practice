import sys

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    t = int(input_data[0])
    results = []
    
    for k in range(t):
        n = int(input_data[2*k + 1])
        s = input_data[2*k + 2]
        
        # We need to check if any b = s[i:j] is a substring of s[0:i] + s[j:n]
        # The easiest case is length of b is 1.
        # Check if any s[i] for 1 <= i < n-1 exists in s[0:i] or s[i+1:n]
        
        found = False
        # Characters available in the potential 'a' and 'c' parts
        # If any character in the middle exists at the ends or is repeated
        prefix_chars = set()
        prefix_chars.add(s[0])
        
        suffix_counts = {}
        for i in range(1, n):
            suffix_counts[s[i]] = suffix_counts.get(s[i], 0) + 1
            
        for i in range(1, n - 1):
            char_b = s[i]
            # Remove from suffix as we move the boundary
            suffix_counts[char_b] -= 1
            if suffix_counts[char_b] == 0:
                del suffix_counts[char_b]
                
            if char_b in prefix_chars or char_b in suffix_counts:
                found = True
                break
            
            prefix_chars.add(char_b)
            
        results.append("Yes" if found else "No")
        
    sys.stdout.write("\n".join(results) + "\n")

solve()
